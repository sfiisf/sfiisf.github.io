实现

class Solution {
static constexpr int mod = 1e9 + 7;
public:
    int numTilings(int n) {
        vector<int> f(n + 1), g(n + 1), h(n + 1);
        f[0] = 1;
        for (int i = 1; i <= n; i++) {
            f[i] = (1ll * f[i - 1] + g[i - 1] + h[i - 1]) % mod;
            if (i > 1) {
                f[i] = (f[i] + f[i - 2]) % mod;
                g[i] = (f[i - 2] + h[i - 1]) % mod;
                h[i] = (f[i - 2] + g[i - 1]) % mod;
            }
        }
        return f[n];
    }
};
